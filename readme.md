Create the database table in Supabase

Open the Supabase SQL Editor and run:

create table president (
  id bigint generated by default as identity primary key,
  firstname text not null,
  lastname text not null,
  birthdate date,
  inserted_at timestamp with time zone default timezone('utc'::text, now()) not null,
  updated_at timestamp with time zone default timezone('utc'::text, now()) not null,
  constraint president_firstname_lastname_unique unique (firstname, lastname)
);


insert into president (firstname, lastname, birthdate)
values
  ('George', 'Washington', '1732-02-22'),
  ('John', 'Adams', '1735-10-30'),
  ('Thomas', 'Jefferson', '1743-04-13'),
  ('James', 'Madison', '1751-03-16'),
  ('James', 'Monroe', '1758-04-28');
